# Venus Protocol API Documentation

## Overview

The Venus Protocol API provides access to pool and market data on BNB Chain. This document outlines the key endpoints and response structures.

## Base URL

```
https://api.venus.io
```

## Endpoints

### Get Pools

Retrieves information about Venus lending pools and their markets.

```
GET /pools?chainId=56
```

#### Parameters

| Parameter | Type   | Required | Description                 |
| --------- | ------ | -------- | --------------------------- |
| chainId   | number | Yes      | Chain ID (56 for BNB Chain) |

#### Response Structure

```typescript
interface PoolsResponse {
  limit: number; // Number of results per page
  page: number; // Current page number
  total: number; // Total number of pools
  result: Pool[]; // Array of pool objects
}

interface Pool {
  address: string; // Pool contract address
  chainId: string; // Chain ID
  name: string; // Pool name
  description: string | null; // Pool description
  priceOracleAddress: string; // Oracle contract address
  closeFactorMantissa: string; // Close factor (scaled by 1e18)
  liquidationIncentiveMantissa: string; // Liquidation incentive (scaled by 1e18)
  minLiquidatableCollateralMantissa: string; // Min liquidatable collateral (scaled by 1e18)
  markets: Market[]; // Array of markets in the pool
}

interface Market {
  address: string; // Market contract address
  chainId: string; // Chain ID
  symbol: string; // Market symbol
  name: string; // Market name
  underlyingAddress: string; // Underlying token address
  underlyingName: string; // Underlying token name
  underlyingSymbol: string; // Underlying token symbol
  underlyingDecimal: number; // Underlying token decimals

  // Interest rates
  borrowRatePerBlock: string; // Borrow rate per block
  supplyRatePerBlock: string; // Supply rate per block
  borrowApy: string; // Annual borrow rate
  supplyApy: string; // Annual supply rate

  // Market state
  exchangeRateMantissa: string; // Exchange rate (scaled)
  underlyingPriceMantissa: string; // Token price (scaled)
  totalBorrowsMantissa: string; // Total borrows
  totalSupplyMantissa: string; // Total supply
  cashMantissa: string; // Available cash
  totalReservesMantissa: string; // Total reserves

  // Configuration
  reserveFactorMantissa: string; // Reserve factor
  collateralFactorMantissa: string; // Collateral factor
  supplyCapsMantissa: string; // Supply cap
  borrowCapsMantissa: string; // Borrow cap

  // Market metrics
  borrowerCount: number; // Number of borrowers
  supplierCount: number; // Number of suppliers
  liquidityCents: string; // Liquidity in cents
  tokenPriceCents: string; // Token price in cents

  // Status
  pausedActionsBitmap: number; // Paused actions bitmap
  isListed: boolean; // Market listing status

  // Timestamps
  createdAt: string; // Creation timestamp
  updatedAt: string; // Last update timestamp

  // Rewards
  rewardsDistributors: RewardDistributor[]; // Reward distributors
}

interface RewardDistributor {
  id: string; // Unique identifier
  marketAddress: string; // Market address
  rewardTokenAddress: string; // Reward token address
  chainId: string; // Chain ID
  supplySpeed: string; // Supply reward speed
  borrowSpeed: string; // Borrow reward speed
  priceMantissa: string; // Reward token price
  rewardType: string; // Reward type
  isActive: boolean; // Active status
  createdAt: string; // Creation timestamp
  updatedAt: string; // Last update timestamp
}
```
